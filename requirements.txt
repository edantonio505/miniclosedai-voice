# MiniClosedAI voice service — pinned dependency manifest.
#
# Versions mirror the proven-working install at /home/edgar/Desktop/BCP_stuff/env
# (Chatterbox running on the same GB10 GPU). Chatterbox-tts itself is installed
# in the Dockerfile with --no-deps; this file pins the sub-deps it actually
# USES at runtime (which differ from its hard-pinned metadata).
#
# Torch / torchaudio are NOT listed here — the Dockerfile pre-installs them
# from pytorch.org's cu130 index in a separate layer so this resolution layer
# can be Docker-cached independently of app dep changes.

# ── Web layer
fastapi==0.115.6
uvicorn[standard]==0.34.0
python-multipart==0.0.20
httpx==0.28.1

# ── Audio I/O
# deepfilternet 0.5.6 hard-pins numpy<2.0. Chatterbox is happy with either
# (BCP_stuff/env uses 2.x but works on 1.26 too), so we satisfy DF here.
numpy>=1.24,<2
soundfile==0.13.1
# av<13 because aiortc still pins to it.
av>=11,<13

# ── ASR: HuggingFace Whisper on GPU (transformers pipeline).
# Versions match BCP_stuff/env's working set.
transformers==4.49.0
accelerate==1.12.0
huggingface-hub==0.36.2
tokenizers==0.21.0
safetensors==0.4.5

# ── Chatterbox TTS sub-deps (chatterbox-tts==0.1.6 is installed in the
# Dockerfile with --no-deps to skip its torch==2.6 hard pin; these are the
# runtime deps it actually needs).
librosa==0.11.0
s3tokenizer==0.3.0
conformer==0.3.2
resemble-perth==1.0.1
diffusers==0.37.0
omegaconf>=2.3,<3

# Chinese / Japanese support packages — chatterbox imports them at startup

# ── WebRTC: FastRTC owns the silero VAD + aiortc plumbing for /call mode.
# [vad] extra pulls silero-vad + onnxruntime so ReplyOnPause works at runtime.
fastrtc[vad]==0.0.34
# Gradio pulled in transitively by fastrtc — pin to the pair that resolves
# in one pass instead of backtracking through 60 MB wheels.
gradio==5.17.0
gradio-client==1.7.1

# ── Misc

# resemble-perth (chatterbox's watermarker) uses pkg_resources which was

setuptools<81
